import requests
from dotenv import load_dotenv
from pathlib import Path
import os

# Load .env from the same folder as this script
load_dotenv(Path(__file__).parent / ".env")

base     = os.getenv("API_BASE")
token    = os.getenv("API_TOKEN")
agent_id = os.getenv("AGENT_ID")

if not all([base, token, agent_id]):
    raise ValueError("Missing required env vars: API_BASE, API_TOKEN, AGENT_ID")

# Base directory for files — same folder as this script
FILES_DIR = Path(__file__).parent / "files"

def upload_file(file_name: str):
    file_path = FILES_DIR / file_name
    if not file_path.exists():
        raise FileNotFoundError(f"File not found: {file_path}")

    with open(file_path, "rb") as file_handle:
        resp = requests.post(
            f"{base}/upload-and-process",
            headers={"Authorization": f"Bearer {token}"},
            files={"files": file_handle},
            data={"agent_session_id": agent_id, "agent_name": "My Agent"},
        )
        resp.raise_for_status()
        return resp.json()


# Upload each file individually
upload_file("file1.txt")
upload_file("file2.txt")
upload_file("file3.txt")
