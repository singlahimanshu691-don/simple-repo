@app.post("/api/run/mcp")
def run_mcp():
    global mcp_process

    # If already running, don't start again
    # if mcp_process is not None and mcp_process.poll() is None:
    #     return {"message": "MCP server already running"}

    script = Path("/app/scripts/start_mcp.sh")
    if not script.exists():
        raise HTTPException(status_code=404, detail="start_mcp.sh not found")

    logger.info("[MCP] Starting MCP server via start_mcp.sh")
    mcp_process = subprocess.Popen(
        ["bash", str(script)],
        cwd="/app",
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE
    )
    logger.info(f"[MCP] Process started with PID {mcp_process.pid}")

    # Give the server a moment to write mcp_url.txt
    time.sleep(3)

    return {"message": f"MCP server started (PID {mcp_process.pid})"}


# ─────────── Notification poller ────────────────────────────────────────────
@app.get("/api/mcp/url")
def get_mcp_url():
    url_file = Path("/app/agent_inputs/mcp_server_url.txt")
    if not url_file.exists():
        raise HTTPException(status_code=404, detail="mcp_url.txt not found yet")
    url = url_file.read_text().strip()
    if not url:
        raise HTTPException(status_code=404, detail="mcp_url.txt is empty")
    logger.info(f"[MCP] URL fetched: {url}")
    return {"url": url}
