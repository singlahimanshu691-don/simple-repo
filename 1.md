async function runMCP() {
      // Step 1 — start the MCP server
      const result = await api('POST', '/api/run/mcp', 'fb-mcp', '✓ MCP server starting...');
      if (!result) return;

      // Step 2 — fetch the URL after server starts
      showFeedback('fb-mcp', 'info', '<span class="loading-spinner"></span> Fetching MCP URL...');

      // Poll until mcp_url.txt is ready (max 10 tries)
      let url = null;
      for (let i = 0; i < 10; i++) {
        await new Promise(r => setTimeout(r, 1500)); // wait 1.5s between tries
        try {
          const res = await fetch('/api/mcp/url');
          if (res.ok) {
            const data = await res.json();
            url = data.url;
            break;
          }
        } catch (e) { }
      }

      if (url) {
        showFeedback('fb-mcp', 'ok', `✓ MCP Server running at: <strong>${url}</strong>`);
        document.getElementById('mcp-url-display').textContent = url;
        document.getElementById('mcp-url-display').style.display = 'block';
      } else {
        showFeedback('fb-mcp', 'err', '✗ MCP server started but URL not written yet — check logs');
      }
    }
