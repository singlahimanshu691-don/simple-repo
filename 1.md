<!-- Repeat this block for each title -->
<div style="display:flex; align-items:center; gap:8px;">
  <span>Your Title</span>
  <div style="position:relative; display:inline-block;">

    <i class="fa-solid fa-circle-info info-icon"
       data-target="pop-1"                        <!-- unique ID -->
       style="font-size:15px; cursor:pointer;">
    </i>

    <div id="pop-1" class="info-popup"            <!-- matches data-target -->
      style="display:none; position:absolute; top:24px;
             left:50%; transform:translateX(-50%);
             background:white; border:1px solid #ddd;
             border-radius:10px; padding:10px; width:220px; z-index:10;">
      <img src="your-image.jpg" style="width:100%;" />
    </div>

  </div>
</div>


// One function handles ALL icons — no duplication
document.querySelectorAll('.info-icon').forEach(function(icon) {
  icon.addEventListener('click', function(e) {
    e.stopPropagation();
    const popup  = document.getElementById(icon.getAttribute('data-target'));
    const isOpen = popup.style.display === 'block';

    // close all first
    document.querySelectorAll('.info-popup').forEach(p => p.style.display = 'none');

    // open this one if it was closed
    if (!isOpen) popup.style.display = 'block';
  });
});

// click outside closes all
document.addEventListener('click', function() {
  document.querySelectorAll('.info-popup').forEach(p => p.style.display = 'none');
});
