// ── API helpers ────────────────────────────────────────────────────────────
async function api(method, path, fbId, successMsg, isDevFb) {
  try {
    showFeedback(fbId, 'info', '<span class="spinner"></span> working…', isDevFb);
    const r = await fetch(path, { method });

  }
}

}

// ── Render service health ──────────────────────────────────────────────────
function renderHealth(data) {
  const container = document.getElementById('serviceCards');
  let healthyCount = 0, faultyCount = 0, totalCount = 0;

  container.innerHTML = Object.entries(data).map(([name, svc], idx) => {
    totalCount++;
    const st = (svc.status || 'UNKNOWN').toLowerCase();
    const statusClass = st === 'healthy' ? 'healthy' : st === 'unreachable' ? 'unreachable' : 'unhealthy';
    const badgeClass = st === 'healthy' ? 'badge-healthy' : st === 'unreachable' ? 'badge-unreachable' : 'badge-unhealthy';
    if (st === 'healthy') healthyCount++; else faultyCount++;

    const issues = (svc.issues || []).slice(0, 2).map(i =>
      `<div class="svc-issue">${i}</div>`).join('');

    let flagsHtml = '';
    if (svc.service_fault_active) flagsHtml += '<span class="fault-flag">SVC FAULT</span>';
    if (svc.db_fault_active) flagsHtml += '<span class="fault-flag">DB FAULT</span>';
    if (svc.silent_fault_active) flagsHtml += '<span class="fault-flag">SILENT FAULT</span>';
    if (svc.fault_active) flagsHtml += '<span class="fault-flag">FAULT</span>';
    const flagsSection = flagsHtml ? `<div class="fault-flags">${flagsHtml}</div>` : '';

    const shortName = name.replace('-service', '');
    return `<div class="service-card ${statusClass}" style="animation-delay:${idx * 40}ms">
      <div class="svc-row">
        <div class="svc-name">${shortName}</div>
        <div class="svc-badge ${badgeClass}">${(svc.status || 'UNKNOWN').toUpperCase()}</div>
      </div>
      ${issues}${flagsSection}
    </div>`;
  }).join('');

  // Summary
  document.getElementById('summHealthy').textContent = healthyCount;
  document.getElementById('summFaulty').textContent = faultyCount;
  document.getElementById('svc-count').textContent = `${totalCount} total`;

  const dot = document.getElementById('platformDot');
  const txt = document.getElementById('platformText');
  if (healthyCount === totalCount) {
    dot.className = 'status-dot'; txt.textContent = 'ALL SYSTEMS OPERATIONAL';
  } else if (healthyCount >= totalCount * 0.5) {
    dot.className = 'status-dot degraded'; txt.textContent = 'PLATFORM DEGRADED';
  } else {
    dot.className = 'status-dot incident'; txt.textContent = 'INCIDENT ACTIVE';
  }
}

// ── Render graph ───────────────────────────────────────────────────────────
function renderGraph(data) {
  const svcCount = (data.services || []).length;
  const vehicleTotal = (data.vehicles || []).reduce((a, v) => a + Number(v.count || 0), 0);
  const jiraCount = data.jira_count || 0;

  document.getElementById('statServices').textContent = svcCount;
  document.getElementById('statVehicles').textContent = vehicleTotal;
  document.getElementById('summVehicles').textContent = vehicleTotal;

  // Two ticket displays
  const jiraEl = document.getElementById('statJira');
  jiraEl.textContent = jiraCount;
  jiraEl.className = 'stat-val' + (jiraCount > 0 ? ' highlight' : '');

  const ticketEl = document.getElementById('statTickets');
  ticketEl.textContent = jiraCount > 0 ? `${jiraCount} tickets` : '—';
  ticketEl.className = 'stat-val' + (jiraCount > 0 ? ' highlight' : '');
  ticketEl.style.fontSize = '11px';
  ticketEl.style.fontFamily = 'var(--mono)';

  updateSeedIndicators(jiraCount);

  const nodes = document.getElementById('graphNodes');
  const allNodes = [
    ...(data.services || []).map(s => ({ ...s, nodeType: 'Service' })),
    ...(data.datastores || []).map(d => ({ ...d, nodeType: 'Datastore' })),
    ...(data.broker || []).map(b => ({ ...b, nodeType: 'Broker' })),
  ];

  nodes.innerHTML = allNodes.map(n => {
    const st = (n.state || 'UNKNOWN').toLowerCase();
    const dotCls = st === 'healthy' ? 'green' : st === 'unhealthy' ? 'red' : 'amber';
    const badgeCls = st === 'healthy' ? 'badge-healthy' : 'badge-unhealthy';
    return `<div class="graph-node">
      <div class="gn-info">
        <div class="gn-dot ${dotCls}"></div>
        <div>
          <span class="gn-name">${n.name}</span>
          <span class="gn-type"> [${n.nodeType}]</span>
        </div>
      </div>
      <div class="svc-badge ${badgeCls}" style="font-size:7.5px">${n.state || 'UNKNOWN'}</div>
    </div>`;
  }).join('');

  const rca = data.rca_candidates || [];
  const rcaSection = document.getElementById('rcaSection');
  if (rca.length > 0) {
    rcaSection.style.display = 'block';
    document.getElementById('rcaCandidates').innerHTML = rca.map(r =>
      `<div class="rca-item">${r.name} <span style="color:var(--text3)">[${r.type}]</span></div>`
    ).join('');
  } else {
    rcaSection.style.display = 'none';
  }
}

  if (ind3) { ind3.className = 'seed-dot ' + (hasTickets ? 'tickets' : 'no-tickets'); txt3.textContent = hasTickets ? `✓ ${jiraCount} incident tickets loaded` : 'No history yet — click Load Incident History'; }
}

// ── Render logs ────────────────────────────────────────────────────────────
async function fetchLogs() {
  const service = document.getElementById('logServiceSelect').value;
  try {
    const r = await fetch(`/api/logs/${service}?lines=60`);
    const data = await r.json();
    const viewer = document.getElementById('logViewer');

    document.getElementById('logCount').innerHTML = `
      <span>lines: ${data.total_lines || 0}</span>
      <span class="err-c">errors: ${data.error_count || 0}</span>
      <span class="warn-c">warnings: ${data.warning_count || 0}</span>`;

    const lines = (data.signal_lines || []);
    if (lines.length === 0) {
      viewer.innerHTML = '<div style="color:var(--text3)">No significant log entries.</div>';
      return;
    }

    const errorSet = new Set(data.errors || []);
    const warnSet = new Set(data.warnings || []);

    viewer.innerHTML = lines.map(line => {
      let cls = 'normal';
      if (errorSet.has(line)) cls = 'error';
      else if (warnSet.has(line)) cls = 'warning';
      else if (line.includes('INFO')) cls = 'info';
      const esc = line.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
      return `<div class="log-line ${cls}">${esc}</div>`;
    }).join('');

    viewer.scrollTop = viewer.scrollHeight;
  } catch (e) {
    document.getElementById('logViewer').innerHTML = `<div style="color:var(--red)">Error: ${e.message}</div>`;
  }
}

  }
}
