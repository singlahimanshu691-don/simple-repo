import requests

base = "<api_base>"
token = "<token>"
agent_id = "<agent_id>"
file_path = "/path/to/context.pdf"

with open(file_path, "rb") as file_handle:
    files = {"files": file_handle}
    data = {"agent_session_id": agent_id, "agent_name": "My Agent"}
    resp = requests.post(
        f"{base}/upload-and-process",
        headers={"Authorization": f"Bearer {token}"},
        files=files,
        data=data,
    )
    resp.raise_for_status()
    print(resp.json())
