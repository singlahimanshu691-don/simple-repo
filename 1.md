echo ""
echo "── Starting Cloudflare Tunnel ───────────────────────"

# Kill any existing cloudflared tunnel
pkill -f "cloudflared tunnel" 2>/dev/null || true
sleep 1

CF_LOG="/tmp/autoops_cloudflare.log"

if [ -f "$CF_CONFIG" ] && ! grep -q "<YOUR_TUNNEL_ID>" "$CF_CONFIG"; then
    # Use config file if it has been filled in
    nohup cloudflared tunnel --config "$CF_CONFIG" run > "$CF_LOG" 2>&1 &
    CF_PID=$!
    echo "  Using config   : $CF_CONFIG"
else
    # Fallback: quick tunnel (no domain, generates random URL)
    echo -e "${YELLOW}  Config not set up — using quick tunnel (random URL)${NC}"
    echo "  See README for permanent domain setup."
    nohup cloudflared tunnel --url http://localhost:8080 > "$CF_LOG" 2>&1 &
    CF_PID=$!
fi

echo "  Cloudflare PID : $CF_PID"
echo "  Log file       : $CF_LOG"

# Wait for tunnel URL to appear in log
echo "  Waiting for tunnel URL..."
TUNNEL_URL=""
for i in {1..20}; do
    TUNNEL_URL=$(grep -o 'https://[a-zA-Z0-9._-]*\.trycloudflare\.com' "$CF_LOG" 2>/dev/null | head -1)
    if [ -n "$TUNNEL_URL" ]; then
        break
    fi
    sleep 1
done

# autoops-poc/cloudflare/config.yml
# Cloudflare Tunnel configuration for AutoOps-AI MCP Server
#
# This exposes ONLY the MCP server (port 8080) — one tunnel, one URL.
# The MCP server internally calls all five microservices on localhost.
#
# Replace the following before use:
#   <YOUR_TUNNEL_ID>   - from: cloudflared tunnel create autoops-poc
#   <YOUR_DOMAIN>      - your domain registered in Cloudflare
#   <YOUR_ACCOUNT_ID>  - from Cloudflare dashboard

tunnel: <YOUR_TUNNEL_ID>
credentials-file: /root/.cloudflared/<YOUR_TUNNEL_ID>.json

ingress:
  - hostname: autoops-mcp.<YOUR_DOMAIN>
    service: http://localhost:8080
    originRequest:
      connectTimeout: 30s
      tcpKeepAlive: 30s
      noTLSVerify: false

  # Catch-all — required by cloudflared
  - service: http_status:404
