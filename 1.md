FROM python:3.11-slim
WORKDIR /app

# Install Docker CLI + curl (for cloudflared download)
RUN apt-get update && apt-get install -y docker.io curl && rm -rf /var/lib/apt/lists/*

# Install cloudflared
RUN curl -fsSL https://github.com/cloudflare/cloudflared/releases/latest/download/cloudflared-linux-amd64 \
    -o /usr/local/bin/cloudflared && chmod +x /usr/local/bin/cloudflared

COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

COPY main.py .
COPY dashboard.html .
COPY entrypoint.sh .
RUN chmod +x entrypoint.sh

EXPOSE 8090

CMD ["./entrypoint.sh"]


#!/bin/bash
set -e

CF_LOG="/tmp/cloudflare_tunnel.log"
MCP_PORT="${MCP_PORT:-8080}"

echo "── Starting Cloudflare Quick Tunnel ─────────────────"
# Quick tunnel — no token, no config, generates a random trycloudflare.com URL
nohup cloudflared tunnel --url "http://localhost:${MCP_PORT}" > "$CF_LOG" 2>&1 &
CF_PID=$!
echo "  Cloudflare PID : $CF_PID"
echo "  Log file       : $CF_LOG"

# Wait up to 20s for the tunnel URL to appear
echo "  Waiting for tunnel URL..."
TUNNEL_URL=""
for i in {1..20}; do
    TUNNEL_URL=$(grep -o 'https://[a-zA-Z0-9._-]*\.trycloudflare\.com' "$CF_LOG" 2>/dev/null | head -1)
    if [ -n "$TUNNEL_URL" ]; then
        echo ""
        echo "  ✅ Tunnel URL: $TUNNEL_URL"
        echo ""
        break
    fi
    sleep 1
done

if [ -z "$TUNNEL_URL" ]; then
    echo "  ⚠️  Tunnel URL not detected after 20s — check $CF_LOG"
fi

echo "── Starting Dashboard (uvicorn) ──────────────────────"
exec uvicorn main:app --host 0.0.0.0 --port 8090 --log-level warning
