async function injectFault(service) {
      const fbId = service.includes('mqtt') ? 'fb-uc1b' : 'fb-uc2';
      await api('POST', `/api/fault/inject/${encodeURIComponent(service)}`, fbId, `✓ Fault injected into ${service}`);
      setTimeout(refreshAll, 1500);
    }
    async function injectDbFault() {
      const fbId = document.getElementById('uc2').classList.contains('active') ? 'fb-uc2' : 'fb-uc3';
      await api('POST', '/api/fault/inject-db', fbId, '✓ SQLite DB fault injected into diagnostics-service');
      setTimeout(refreshAll, 1500);
    }
    async function injectSilentFault() {
      await api('POST', '/api/fault/inject-silent', 'fb-uc4', '✓ Silent fault active — watch logs panel (diagnostics-service)');
      document.getElementById('logServiceSelect').value = 'diagnostics-service';
      setTimeout(fetchLogs, 4000);
    }
