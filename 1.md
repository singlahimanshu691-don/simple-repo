#!/bin/bash
# start_mcp.sh
# Starts the FastMCP server and Cloudflare Tunnel in one command.
#
# Usage:
#   chmod +x scripts/start_mcp.sh
#   ./scripts/start_mcp.sh
#
# Prerequisites:
#   1. docker compose up -d  (PoC services must be running)
#   2. python scripts/seed_neo4j.py
#   3. cloudflared installed and tunnel created (see README)

set -e

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_DIR="$(dirname "$SCRIPT_DIR")"
MCP_DIR="$PROJECT_DIR/mcp_server"
CF_CONFIG="$PROJECT_DIR/cloudflare/config.yml"

GREEN='\033[0;32m'
RED='\033[0;31m'
YELLOW='\033[1;33m'
NC='\033[0m'

echo ""
echo "╔══════════════════════════════════════════════╗"
echo "║    AutoOps-AI — MCP + Cloudflare Startup     ║"
echo "╚══════════════════════════════════════════════╝"
echo ""

# ── Check prerequisites ───────────────────────────────────────────────────────

echo "── Checking prerequisites ───────────────────────────"

# Check Python
if ! command -v python3 &>/dev/null; then
    echo -e "${RED}ERROR: python3 not found${NC}"
    exit 1
fi
echo -e "  ${GREEN}✓${NC}  python3 found"

# Check fastmcp
if ! python3 -c "import fastmcp" 2>/dev/null; then
    echo -e "${YELLOW}  Installing MCP server dependencies...${NC}"
    # pip install -r "$MCP_DIR/requirements.txt"
fi
echo -e "  ${GREEN}✓${NC}  fastmcp installed"

# Check cloudflared
if ! command -v cloudflared &>/dev/null; then
    echo -e "${RED}ERROR: cloudflared not found.${NC}"
    echo "  Install: curl -L https://github.com/cloudflare/cloudflared/releases/latest/download/cloudflared-linux-amd64.deb -o cloudflared.deb && sudo dpkg -i cloudflared.deb"
    exit 1
fi
echo -e "  ${GREEN}✓${NC}  cloudflared found"

# Check docker services
echo ""
echo "── Checking Docker services ─────────────────────────"
ALL_UP=true
for port in 8001 8002 8003 8004 8005 7687; do
    if nc -z localhost $port 2>/dev/null; then
        echo -e "  ${GREEN}✓${NC}  port $port is open"
    else
        echo -e "  ${RED}✗${NC}  port $port not open"
        ALL_UP=false
    fi
done

if [ "$ALL_UP" = false ]; then
    echo ""
    echo -e "${YELLOW}WARNING: Some services not running.${NC}"
    echo "Run: docker compose up -d && python scripts/seed_neo4j.py"
    echo ""
    read -p "Continue anyway? (y/N) " -n 1 -r
    echo
    if [[ ! $REPLY =~ ^[Yy]$ ]]; then
        exit 1
    fi
fi

# ── Start FastMCP server ──────────────────────────────────────────────────────

echo ""
echo "── Starting FastMCP server ──────────────────────────"

# Kill any existing MCP server
pkill -f "mcp_server/server.py" 2>/dev/null || true
sleep 1

MCP_LOG="/tmp/autoops_mcp.log"
nohup python3 "$MCP_DIR/server.py" > "$MCP_LOG" 2>&1 &
MCP_PID=$!
echo "  MCP server PID : $MCP_PID"
echo "  Log file       : $MCP_LOG"

# Wait for MCP server to be ready
echo "  Waiting for MCP server to start..."
for i in {1..10}; do
    if curl -s http://localhost:8080/mcp > /dev/null 2>&1; then
        echo -e "  ${GREEN}✓${NC}  MCP server is ready"
        break
    fi
    sleep 1
    if [ $i -eq 10 ]; then
        echo -e "${RED}ERROR: MCP server did not start in 10s${NC}"
        echo "Check logs: cat $MCP_LOG"
        exit 1
    fi
done

# ── Start Cloudflare Tunnel ───────────────────────────────────────────────────

echo ""
echo "── Starting Cloudflare Tunnel ───────────────────────"

# Kill any existing cloudflared tunnel
pkill -f "cloudflared tunnel" 2>/dev/null || true
sleep 1

CF_LOG="/tmp/autoops_cloudflare.log"

if [ -f "$CF_CONFIG" ] && ! grep -q "<YOUR_TUNNEL_ID>" "$CF_CONFIG"; then
    # Use config file if it has been filled in
    nohup cloudflared tunnel --config "$CF_CONFIG" run > "$CF_LOG" 2>&1 &
    CF_PID=$!
    echo "  Using config   : $CF_CONFIG"
else
    # Fallback: quick tunnel (no domain, generates random URL)
    echo -e "${YELLOW}  Config not set up — using quick tunnel (random URL)${NC}"
    echo "  See README for permanent domain setup."
    nohup cloudflared tunnel --url http://localhost:8080 > "$CF_LOG" 2>&1 &
    CF_PID=$!
fi

echo "  Cloudflare PID : $CF_PID"
echo "  Log file       : $CF_LOG"

# Wait for tunnel URL to appear in log
echo "  Waiting for tunnel URL..."
TUNNEL_URL=""
for i in {1..20}; do
    TUNNEL_URL=$(grep -o 'https://[a-zA-Z0-9._-]*\.trycloudflare\.com' "$CF_LOG" 2>/dev/null | head -1)
    if [ -n "$TUNNEL_URL" ]; then
        break
    fi
    sleep 1
done

echo ""
echo "══════════════════════════════════════════════════════"
echo "  MCP Server URL  : http://localhost:8080/mcp"
if [ -n "$TUNNEL_URL" ]; then
    echo -e "  ${GREEN}Public MCP URL  : $TUNNEL_URL/mcp${NC}"
    echo ""
    echo "  Configure your AI agent with:"
    echo "    MCP Server URL : $TUNNEL_URL/mcp"
    echo "    Transport      : streamable-http"

    # Save URL to file for reference
    echo "$TUNNEL_URL/mcp" > "$PROJECT_DIR/agent_inputs/mcp_server_url.txt"
    echo "  Saved to       : agent_inputs/mcp_server_url.txt"
else
    echo -e "${YELLOW}  Public URL not detected yet — check: cat $CF_LOG${NC}"
fi
echo "══════════════════════════════════════════════════════"
echo ""
echo "  To stop everything:"
echo "    pkill -f 'mcp_server/server.py'"
echo "    pkill -f 'cloudflared tunnel'"
echo ""
echo "  To view logs:"
echo "    tail -f $MCP_LOG"
echo "    tail -f $CF_LOG"
echo ""
