def upload_file(file_name: str):
    file_path = FILES_DIR / file_name
    if not file_path.exists():
        raise FileNotFoundError(f"File not found: {file_path}")

    with open(file_path, "rb") as file_handle:
        resp = requests.post(
            f"{base}/upload-and-process",
            headers={"Authorization": f"Bearer {token}"},
            files={"files": file_handle},
            data={"agent_session_id": agent_id, "agent_name": "My Agent"},
        )
        resp.raise_for_status()
        return resp.json()

def scenario_1():
    """
    UC1 (Part B) — mqtt-bridge fault causing mass vehicle disconnect.

    Injects a software fault into mqtt-bridge-service. The container
    keeps running but /health returns HTTP 503. Vehicles disconnect
    because the bridge can no longer maintain MQTT connectivity.

    Graph RCA: mqtt-bridge-service UNHEALTHY, no unhealthy upstream deps
               → root cause identified, downstream: vehicle-telemetry cascade
    Remediation: REM-002 — clear_service_fault(mqtt-bridge-service)
    """
    print(c("BOLD", "\n🔴  SCENARIO 1 — UC1: mqtt-bridge fault (mass vehicle disconnect)"))
    print("─" * 65)
    print("Injecting fault into: mqtt-bridge-service")
    print("Expected: mqtt-bridge-service UNHEALTHY")
    print("          vehicle-telemetry-service cascades to UNHEALTHY")
    print("          vehicles appear offline")
    print("RCA target: mqtt-bridge-service (no unhealthy deps)")
    print()

    call("POST", f"{SERVICES['mqtt-bridge']}/fault/inject",
         "Inject fault → mqtt-bridge-service")

    # Poll all downstream services in strict dependency order so each one
    # detects the upstream fault and writes UNHEALTHY to Neo4j immediately.
    # Order matters — each service checks its direct upstream, so they must
    # be polled top-down: vehicle-telemetry first, then diagnostics, etc.
    print(c("YELLOW", "\nPolling all downstream services to propagate cascade into Neo4j graph..."))
    time.sleep(1)

    downstream_order = [
        "vehicle-telemetry",   # depends on mqtt-bridge        → UNHEALTHY
        "diagnostics",         # depends on vehicle-telemetry  → UNHEALTHY
        "maintenance",         # depends on diagnostics        → UNHEALTHY
        "notification",        # depends on maintenance        → UNHEALTHY
        "dealer-portal",       # depends on notification       → UNHEALTHY
    ]
    for name in downstream_order:
        url = SERVICES[name]
        try:
            httpx.get(f"{url}/health", timeout=4.0)
            print(f"  polled {name}")
        except Exception:
            print(f"  {name} unreachable (expected if cascaded)")
        time.sleep(0.5)

    time.sleep(1)
    print_health("Post-fault health")

    print(c("CYAN", "Agent RCA:"))
    print("  Graph query: unhealthy node with no unhealthy upstream deps")
    print("  → mqtt-bridge-service (Service node)")
    print("  Remediation: REM-002 — clear_service_fault(mqtt-bridge-service)")
    print("\nRun:  python scripts/simulate_failure.py clear  to restore\n")
    # -----------------------------------------------------------------
    #  Upload the context file that documents the failure scenario.
    # -----------------------------------------------------------------

        resp = upload_file("context-2.txt")
        print(c("GREEN", "\n✅ Context file uploaded successfully."))
        print("Server response:", resp)

    except Exception as exc:  # catch‑all for unexpected issues
        print(c("RED", f"\n❌ Unexpected error: {exc}"))
