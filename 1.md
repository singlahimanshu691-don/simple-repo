declare -A SERVICE_PORTS=(
    [vehicle-telemetry]=8001
    [diagnostics]=8002
    [maintenance]=8003
    [notification]=8004
    [dealer-portal]=8005
    [neo4j]=7687
)

for service in "${!SERVICE_PORTS[@]}"; do
    port="${SERVICE_PORTS[$service]}"
    if nc -z "$service" "$port" 2>/dev/null; then
        echo -e "  ${GREEN}✓${NC}  $service:$port is reachable"
    else
        echo -e "  ${RED}✗${NC}  $service:$port not reachable"
        ALL_UP=false
    fi
done
